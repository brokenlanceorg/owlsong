#include"Hedge.hpp"

HedgE::HedgE()
{
HFPRINT( "HedgE::HedgE()" );
  Setup();
} // end default constructor

HedgE::~HedgE()
{
HFPRINT( "HedgE::~HedgE()" );

  if( Fct1The_Hedge_Function != 0 )
    delete Fct1The_Hedge_Function;
} // end destructor

void HedgE::Setup()
{
HFPRINT( "HedgE::Setup()" );

  FzsNext_Set = 0;
  Fct1The_Hedge_Function = 0;
  ldPower_Variable = 0;
  ldModified_Power_Variable = 0;
} // end Setup

HedgE& HedgE::operator + ( FuzzyseT& FzsRHS )
{
HFPRINT( "HedgE::operator + (const FuzzyseT& FzsRHS)" );

  // Make sure that we don't point to ourselves, which could be fatal.
  if ( this == &FzsRHS )
  {
    //HFPRINT << "Caught reference to self!";
    cout << "Caught reference to self!";
    return *this;
}
  FzsNext_Set = &FzsRHS;

  return *this;
} // end operator +
